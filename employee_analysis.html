import pandas as pd
import matplotlib.pyplot as plt

data = {
    "EmployeeID": range(1, 101),
    "Department": ["Sales", "Marketing", "Finance", "HR", "IT"] * 20,
    "Region": ["North", "South", "East", "West", "Central"] * 20,
    "PerformanceScore": [60, 70, 80, 90, 75] * 20
}

df = pd.DataFrame(data)

marketing_count = (df["Department"] == "Marketing").sum()
print("Marketing count:", marketing_count)

plt.figure(figsize=(8, 5))
plt.hist(df["Department"], bins=len(df["Department"].unique()), edgecolor="black")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Frequency")
plt.tight_layout()
plt.savefig("department_histogram.png")
